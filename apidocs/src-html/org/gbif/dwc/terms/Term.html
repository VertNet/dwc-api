001package org.gbif.dwc.terms;
002
003public interface Term {
004
005  /**
006   * The simple term name without any namespace or paths.
007   * For example scientificName.
008   */
009  String simpleName();
010
011  /**
012   * The full qualified term uri including the namespace.
013   * For example http://rs.tdwg.org/dwc/terms/scientificName.
014   */
015  String qualifiedName();
016
017  /**
018   * Lists known alternative names for the same term. These names are frequently found typos or term names that
019   * have been used in past versions of Darwin Core and are still found in data.
020   *
021   * @return known alternative simple names for this term, often null
022   */
023  String[] alternativeNames();
024}
