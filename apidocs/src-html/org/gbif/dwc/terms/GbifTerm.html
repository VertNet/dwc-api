001package org.gbif.dwc.terms;
002
003import java.util.ArrayList;
004import java.util.List;
005
006public enum GbifTerm implements Term, AlternativeNames {
007  // row types
008  Description(GbifTerm.GROUP_ROW_TYPE),
009  Distribution(GbifTerm.GROUP_ROW_TYPE),
010  Identifier(GbifTerm.GROUP_ROW_TYPE),
011  Image(GbifTerm.GROUP_ROW_TYPE, "Images"),
012  Reference(GbifTerm.GROUP_ROW_TYPE, "References"),
013  SpeciesProfile(GbifTerm.GROUP_ROW_TYPE, "SpeciesMiniProfile", "SpeciesInfo", "SpeciesData", "SpeciesFactsheet"),
014  TypesAndSpecimen(GbifTerm.GROUP_ROW_TYPE, "Specimen", "Types", "TypeDesignation"),
015  VernacularName(GbifTerm.GROUP_ROW_TYPE, "VernacularNames", "Vernacular", "Vernaculars"),
016
017  // Dataset properties
018  datasetKey(GbifTerm.GROUP_DATASET),
019  publishingCountry(GbifTerm.GROUP_DATASET),
020
021  // Occurrence properties
022  lastInterpreted(GbifTerm.GROUP_OCCURRENCE),
023  coordinateAccuracy(GbifTerm.GROUP_OCCURRENCE),
024  elevation(GbifTerm.GROUP_OCCURRENCE),
025  elevationAccuracy(GbifTerm.GROUP_OCCURRENCE),
026  depth(GbifTerm.GROUP_OCCURRENCE),
027  depthAccuracy(GbifTerm.GROUP_OCCURRENCE),
028  distanceAboveSurface(GbifTerm.GROUP_OCCURRENCE),
029  distanceAboveSurfaceAccuracy(GbifTerm.GROUP_OCCURRENCE),
030  issue(GbifTerm.GROUP_OCCURRENCE),
031  // experimental Occurrence properties
032  verbatimLabel(GbifTerm.GROUP_OCCURRENCE),
033  infraspecificMarker(GbifTerm.GROUP_OCCURRENCE),
034  // Types and Specimen checklist extension
035  typeDesignatedBy(GbifTerm.GROUP_OCCURRENCE),
036  typeDesignationType(GbifTerm.GROUP_OCCURRENCE),
037
038  // Taxon properties
039  /**
040   * The GBIF backbone key.
041   * taxonID is only used for the source ids similar to occurrenceID
042   */
043  taxonKey(GbifTerm.GROUP_TAXON),
044  kingdomKey(GbifTerm.GROUP_TAXON),
045  phylumKey(GbifTerm.GROUP_TAXON),
046  classKey(GbifTerm.GROUP_TAXON),
047  orderKey(GbifTerm.GROUP_TAXON),
048  familyKey(GbifTerm.GROUP_TAXON),
049  genusKey(GbifTerm.GROUP_TAXON),
050  subgenusKey(GbifTerm.GROUP_TAXON),
051  speciesKey(GbifTerm.GROUP_TAXON),
052  species(GbifTerm.GROUP_TAXON),
053  // experimental Taxon properties
054  canonicalName(GbifTerm.GROUP_TAXON),
055  nameType(GbifTerm.GROUP_TAXON),
056
057  // GBIF Crawling
058  protocol(GbifTerm.GROUP_CRAWLING),
059  lastParsed(GbifTerm.GROUP_CRAWLING),
060  lastCrawled(GbifTerm.GROUP_CRAWLING),
061
062  // Species Profile checklist extension
063  isMarine(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
064  isTerrestrial(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
065  isFreshwater(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
066  isHybrid(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
067  isExtinct(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
068  livingPeriod(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION, "timePeriod"),
069  lifeForm(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
070  ageInDays(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
071  sizeInMillimeter(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
072  massInGram(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION, "weightInGram"),
073
074  // Vernacular Name checklist extension
075  organismPart(GbifTerm.GROUP_VERNACULAR_NAME_EXTENSION),
076  isPlural(GbifTerm.GROUP_VERNACULAR_NAME_EXTENSION),
077  isPreferredName(GbifTerm.GROUP_VERNACULAR_NAME_EXTENSION),
078
079  // Distribution checklist extension
080  appendixCITES(GbifTerm.GROUP_SPECIES_DISTRIBUTION_EXTENSION);
081
082  public static final String NS = "http://rs.gbif.org/terms/1.0/";
083  public static final String PREFIX = "gbif";
084  static final String[] PREFIXES = {NS, PREFIX + ":"};
085  public static final String GROUP_CRAWLING = "Crawling";
086  public static final String GROUP_DATASET = "Dataset";
087  public static final String GROUP_OCCURRENCE = "Occurrence";
088  public static final String GROUP_ROW_TYPE = "RowType";
089  public static final String GROUP_SPECIES_DISTRIBUTION_EXTENSION = "SpeciesDistribution";
090  public static final String GROUP_SPECIES_PROFILE_EXTENSION = "SpeciesProfile";
091  public static final String GROUP_TAXON = "Taxon";
092  public static final String GROUP_VERNACULAR_NAME_EXTENSION = "VernacularName";
093
094  /**
095   * Lists all GBIF term groups.
096   */
097  public static final String[] GROUPS = {GROUP_CRAWLING, GROUP_DATASET, GROUP_OCCURRENCE, GROUP_ROW_TYPE,
098    GROUP_SPECIES_DISTRIBUTION_EXTENSION, GROUP_SPECIES_PROFILE_EXTENSION, GROUP_TAXON,
099    GROUP_VERNACULAR_NAME_EXTENSION};
100
101  /**
102   * Lists all GBIF terms in taxon group.
103   */
104  public static final GbifTerm[] TAXONOMIC_TERMS =
105    {GbifTerm.taxonKey, GbifTerm.kingdomKey, GbifTerm.phylumKey, GbifTerm.classKey, GbifTerm.orderKey,
106      GbifTerm.familyKey, GbifTerm.genusKey, GbifTerm.subgenusKey, GbifTerm.speciesKey, GbifTerm.species,
107      GbifTerm.canonicalName, GbifTerm.nameType};
108
109  private final String groupName;
110  public final String[] normAlts;
111
112  private GbifTerm(String groupName, String... alternatives) {
113    this.groupName = groupName;
114    normAlts = alternatives;
115  }
116
117  /**
118   * The full qualified term uri including the namespace.
119   * For example http://rs.gbif.org/terms/1.0/taxonKey.
120   *
121   * @return full qualified term uri
122   */
123  @Override
124  public String qualifiedName() {
125    return NS + name();
126  }
127
128  /**
129   * The simple term name without a namespace.
130   * For example taxonKey.
131   *
132   * @return simple term name
133   */
134  @Override
135  public String simpleName() {
136    return name();
137  }
138
139  /**
140   * Array of alternative simple names in use for the term.
141   *
142   * @return simple term name
143   */
144  @Override
145  public String[] alternativeNames() {
146    return normAlts;
147  }
148
149  /**
150   * The optional group the term is grouped in.
151   * For example Occurrence, Taxon, etc.
152   */
153  public String getGroup() {
154    return groupName;
155  }
156
157
158  /**
159   * List all terms that belong to a given group.
160   *
161   * @param group the group to list terms for
162   *
163   * @return the list of GBIF terms in the given group
164   */
165  public static List<GbifTerm> listByGroup(String group) {
166    List<GbifTerm> terms = new ArrayList<GbifTerm>();
167    for (GbifTerm t : GbifTerm.values()) {
168      if (t.getGroup().equalsIgnoreCase(group)) {
169        terms.add(t);
170      }
171    }
172    return terms;
173  }
174
175  @Override
176  public String toString() {
177    return PREFIX + ":" + name();
178  }
179
180  /**
181   * @return true if the term is defining a class or row type instead of a property, e.g. VernacularName
182   */
183  public boolean isClass() {
184    return Character.isUpperCase(simpleName().charAt(0));
185  }
