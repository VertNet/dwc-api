085  public static final String NS = "http://rs.gbif.org/terms/1.0/";
086  public static final String PREFIX = "gbif";
087  static final String[] PREFIXES = {NS, PREFIX + ":"};
088
089  public final String[] normAlts;
090
091  GbifTerm(String... alternatives) {
092    normAlts = alternatives;
093  }
094
095  @Override
096  public String qualifiedName() {
097    return NS + name();
098  }
099
100  @Override
101  public String simpleName() {
102    return name();
103  }
104
105  @Override
106  public String[] alternativeNames() {
107    return normAlts;
108  }
109
110  @Override
111  public String toString() {
112    return PREFIX + ":" + name();
113  }
114
115}
