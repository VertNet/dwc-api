001package org.gbif.dwc.terms;
002
003import java.util.ArrayList;
004import java.util.List;
005
006public enum GbifTerm implements Term, AlternativeNames {
007  // row types
008  Description(GbifTerm.GROUP_ROW_TYPE),
009  Distribution(GbifTerm.GROUP_ROW_TYPE),
010  Identifier(GbifTerm.GROUP_ROW_TYPE),
011  Image(GbifTerm.GROUP_ROW_TYPE, "Images"),
012  Reference(GbifTerm.GROUP_ROW_TYPE, "References"),
013  SpeciesProfile(GbifTerm.GROUP_ROW_TYPE, "SpeciesMiniProfile", "SpeciesInfo", "SpeciesData", "SpeciesFactsheet"),
014  TypesAndSpecimen(GbifTerm.GROUP_ROW_TYPE, "Specimen", "Types", "TypeDesignation"),
015  VernacularName(GbifTerm.GROUP_ROW_TYPE, "VernacularNames", "Vernacular", "Vernaculars"),
016  Multimedia(GbifTerm.GROUP_ROW_TYPE),
017
018  // Dataset properties
019  datasetKey(GbifTerm.GROUP_DATASET),
020  publishingCountry(GbifTerm.GROUP_DATASET),
021
022  // Occurrence properties
023  gbifID(GbifTerm.GROUP_OCCURRENCE),
024  lastInterpreted(GbifTerm.GROUP_OCCURRENCE),
025  coordinateAccuracy(GbifTerm.GROUP_OCCURRENCE),
026  elevation(GbifTerm.GROUP_OCCURRENCE),
027  elevationAccuracy(GbifTerm.GROUP_OCCURRENCE),
028  depth(GbifTerm.GROUP_OCCURRENCE),
029  depthAccuracy(GbifTerm.GROUP_OCCURRENCE),
030  distanceAboveSurface(GbifTerm.GROUP_OCCURRENCE),
031  distanceAboveSurfaceAccuracy(GbifTerm.GROUP_OCCURRENCE),
032  issue(GbifTerm.GROUP_OCCURRENCE),
033  // experimental Occurrence properties
034  verbatimLabel(GbifTerm.GROUP_OCCURRENCE),
035  infraspecificMarker(GbifTerm.GROUP_OCCURRENCE),
036  // Types and Specimen checklist extension
037  typeDesignatedBy(GbifTerm.GROUP_OCCURRENCE),
038  typeDesignationType(GbifTerm.GROUP_OCCURRENCE),
039  // booleans for geospatial issues
040  hasCoordinate(GbifTerm.GROUP_OCCURRENCE),
041  hasGeospatialIssues(GbifTerm.GROUP_OCCURRENCE),
042
043  // Taxon properties
044  /**
045   * The GBIF backbone key.
046   * taxonID is only used for the source ids similar to occurrenceID
047   */
048  taxonKey(GbifTerm.GROUP_TAXON),
049  kingdomKey(GbifTerm.GROUP_TAXON),
050  phylumKey(GbifTerm.GROUP_TAXON),
051  classKey(GbifTerm.GROUP_TAXON),
052  orderKey(GbifTerm.GROUP_TAXON),
053  familyKey(GbifTerm.GROUP_TAXON),
054  genusKey(GbifTerm.GROUP_TAXON),
055  subgenusKey(GbifTerm.GROUP_TAXON),
056  speciesKey(GbifTerm.GROUP_TAXON),
057  species(GbifTerm.GROUP_TAXON),
058  // experimental Taxon properties
059  canonicalName(GbifTerm.GROUP_TAXON),
060  nameType(GbifTerm.GROUP_TAXON),
061
062  // GBIF Crawling
063  protocol(GbifTerm.GROUP_CRAWLING),
064  lastParsed(GbifTerm.GROUP_CRAWLING),
065  lastCrawled(GbifTerm.GROUP_CRAWLING),
066
067  // Species Profile checklist extension
068  isMarine(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
069  isTerrestrial(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
070  isFreshwater(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
071  isHybrid(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
072  isExtinct(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
073  livingPeriod(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION, "timePeriod"),
074  lifeForm(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
075  ageInDays(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
076  sizeInMillimeter(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION),
077  massInGram(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION, "weightInGram"),
078
079  // Vernacular Name checklist extension
080  organismPart(GbifTerm.GROUP_VERNACULAR_NAME_EXTENSION),
081  isPlural(GbifTerm.GROUP_VERNACULAR_NAME_EXTENSION),
082  isPreferredName(GbifTerm.GROUP_VERNACULAR_NAME_EXTENSION),
083
084  // Distribution checklist extension
085  appendixCITES(GbifTerm.GROUP_SPECIES_DISTRIBUTION_EXTENSION);
086
087  public static final String NS = "http://rs.gbif.org/terms/1.0/";
088  public static final String PREFIX = "gbif";
089  static final String[] PREFIXES = {NS, PREFIX + ":"};
090  public static final String GROUP_CRAWLING = "Crawling";
091  public static final String GROUP_DATASET = "Dataset";
092  public static final String GROUP_OCCURRENCE = "Occurrence";
093  public static final String GROUP_ROW_TYPE = "RowType";
094  public static final String GROUP_SPECIES_DISTRIBUTION_EXTENSION = "SpeciesDistribution";
095  public static final String GROUP_SPECIES_PROFILE_EXTENSION = "SpeciesProfile";
096  public static final String GROUP_TAXON = "Taxon";
097  public static final String GROUP_VERNACULAR_NAME_EXTENSION = "VernacularName";
098
099  /**
100   * Lists all GBIF term groups.
101   */
102  public static final String[] GROUPS = {GROUP_CRAWLING, GROUP_DATASET, GROUP_OCCURRENCE, GROUP_ROW_TYPE,
103    GROUP_SPECIES_DISTRIBUTION_EXTENSION, GROUP_SPECIES_PROFILE_EXTENSION, GROUP_TAXON,
104    GROUP_VERNACULAR_NAME_EXTENSION};
105
106  /**
107   * Lists all GBIF terms in taxon group.
108   */
109  public static final GbifTerm[] TAXONOMIC_TERMS =
110    {GbifTerm.taxonKey, GbifTerm.kingdomKey, GbifTerm.phylumKey, GbifTerm.classKey, GbifTerm.orderKey,
111      GbifTerm.familyKey, GbifTerm.genusKey, GbifTerm.subgenusKey, GbifTerm.speciesKey, GbifTerm.species,
112      GbifTerm.canonicalName, GbifTerm.nameType};
113
114  private final String groupName;
115  public final String[] normAlts;
116
117  private GbifTerm(String groupName, String... alternatives) {
118    this.groupName = groupName;
119    normAlts = alternatives;
120  }
121
122  /**
123   * The full qualified term uri including the namespace.
124   * For example http://rs.gbif.org/terms/1.0/taxonKey.
125   *
126   * @return full qualified term uri
127   */
128  @Override
129  public String qualifiedName() {
130    return NS + name();
131  }
132
133  /**
134   * The simple term name without a namespace.
135   * For example taxonKey.
136   *
137   * @return simple term name
138   */
139  @Override
140  public String simpleName() {
141    return name();
142  }
143
144  /**
145   * Array of alternative simple names in use for the term.
146   *
147   * @return simple term name
148   */
149  @Override
150  public String[] alternativeNames() {
151    return normAlts;
152  }
153
154  /**
155   * The optional group the term is grouped in.
156   * For example Occurrence, Taxon, etc.
157   */
158  public String getGroup() {
159    return groupName;
160  }
161
162
163  /**
164   * List all terms that belong to a given group.
165   *
166   * @param group the group to list terms for
167   *
168   * @return the list of GBIF terms in the given group
169   */
170  public static List<GbifTerm> listByGroup(String group) {
171    List<GbifTerm> terms = new ArrayList<GbifTerm>();
172    for (GbifTerm t : GbifTerm.values()) {
173      if (t.getGroup().equalsIgnoreCase(group)) {
174        terms.add(t);
175      }
176    }
177    return terms;
178  }
179
180  @Override
181  public String toString() {
182    return PREFIX + ":" + name();
183  }
184
185  /**
186   * @return true if the term is defining a class or row type instead of a property, e.g. VernacularName
187   */
188  public boolean isClass() {
189    return Character.isUpperCase(simpleName().charAt(0));
190  }
