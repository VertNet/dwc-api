001package org.gbif.dwc.terms;
002
003import java.util.HashMap;
004import java.util.Map;
005import java.util.regex.Pattern;
006
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010/**
011 * Simple factory for terms that knows about all ConceptTerms of this library and keeps singletons for
012 * all unknown Term instances.
013 */
014public class TermFactory {
015    private final static Logger LOG = LoggerFactory.getLogger(TermFactory.class);
016    private static final Pattern NON_ALPHA_NUM_PATTERN = Pattern.compile("[^a-zA-Z0-9]+");
017    private final Map<String, Term> terms = new HashMap<String, Term>();
018
019    public TermFactory() {
020        loadKnownTerms();
021    }
022
023    private void loadKnownTerms() {
024        addTerms(DwcTerm.values(), DwcTerm.PREFIXES);
025        addTerms(DcTerm.values(), DcTerm.PREFIXES);
026        addTerms(GbifTerm.values(), GbifTerm.PREFIXES);
027        addTerms(IucnTerm.values(), IucnTerm.PREFIXES);
028    }
029
030    private void addTerms(Term[] terms, String[] prefixes) {
031        for (Term term : terms) {
032            addTerm(term.simpleName(), term);
033            addTerm(term.qualifiedName(), term);
034            for (String pre : prefixes) {
035                addTerm(pre + term.simpleName(), term);
036            }
037            // alt names
038            for (String alt : term.alternativeNames()) {
039                addTerm(alt, term);
040                for (String pre : prefixes) {
041                    addTerm(pre + alt, term);
042                }
043            }
044
045        }
046    }
047
048    private void addTerm(String key, Term term) {
049        if (key == null || key.trim().isEmpty()) {
050            return;
051        }
052        key = normaliseTerm(key);
053        if (terms.containsKey(key)) {
054            Term t1 = terms.get(key);
055            if (!t1.equals(term)) {
056                LOG.warn("Terms {} and {} are both known as \"{}\". Keeping only {}", terms.get(key), term, key, terms.get(key));
057            }
058        } else {
059            terms.put(key, term);
060        }
061    }
062
063    /**
064     * @return a purely alphanumerical, lower cased term with all other characters replaced
065     */
066    public static String normaliseTerm(String term) {
067        return NON_ALPHA_NUM_PATTERN.matcher(term).replaceAll("").toLowerCase();
068    }
069
070    public Term findTerm(String termName) {
071        if (termName == null || termName.trim().isEmpty()) {
072            return null;
073        }
074        // normalise terms
075        Term term;
076        String normTermName = normaliseTerm(termName);
077        // try known terms first
078        if (terms.containsKey(normTermName)) {
079            return terms.get(normTermName);
080        } else {
081            // create new term instance
082            int pos = termName.lastIndexOf("/");
083            String tmpName = pos == -1 ? "" : termName.substring(pos + 1);
084            term = new UnknownTerm(termName, tmpName);
085            addTerm(normTermName, term);
086            addTerm(termName, term);
087            addTerm(tmpName, term);
088        }
089        return term;
090    }
091
092}
