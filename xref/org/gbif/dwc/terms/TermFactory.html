10  /**
11   * Simple, threadsafe factory for terms that knows about all ConceptTerms of this library and keeps singletons for
12   * all unknown Term instances.
13   */
14  public class TermFactory {

16    private static final Logger LOG = LoggerFactory.getLogger(TermFactory.class);
17    private static final Pattern NON_ALPHA_NUM_PATTERN = Pattern.compile("[^a-zA-Z0-9#-]+");
18    private static final String UNKNOWN_NAMESPACE = "http://unknown.org/";
19    private static TermFactory singleton;
20    private static boolean initialized = false;
21    private static final Object LOCK = new Object();
22  
23    private final Map<String, Term> terms = new HashMap<String, Term>();

25    public static TermFactory instance() {
26      if (initialized) {
27        return singleton;
28      }
29  
30      synchronized (LOCK) {
31        if (singleton == null) {
32          LOG.debug("Building new TermFactory instance");
33          singleton = new TermFactory();
34          singleton.loadKnownTerms();
35          initialized = true;
36        }
37      }
38  
39      return singleton;
40    }

42    private TermFactory() {
43    }

45    private void loadKnownTerms() {
46      addTerms(DwcTerm.values(), DwcTerm.PREFIXES);
47      addTerms(DcTerm.values(), DcTerm.PREFIXES);
48      addTerms(GbifTerm.values(), GbifTerm.PREFIXES);
49      addTerms(GbifInternalTerm.values(), new String[0]);
50      addTerms(IucnTerm.values(), IucnTerm.PREFIXES);
51  
52    }
53  
54    private <T extends Term & AlternativeNames> void addTerms(T[] terms, String[] prefixes) {
55      for (T term : terms) {
56        addTerm(term.simpleName(), term);
57        addTerm(term.qualifiedName(), term);
58        for (String pre : prefixes) {
59          addTerm(pre + term.simpleName(), term);
60        }
61        // also index alt names
62        for (String alt : term.alternativeNames()) {
63          addTerm(alt, term);
64          for (String pre : prefixes) {
65            addTerm(pre + alt, term);
66          }
67        }
68      }
69    }
70  
71    public void addTerm(String key, Term term) {
72      if (key == null || key.trim().isEmpty()) {
73        return;
74      }
75      key = normaliseTerm(key);
76      if (terms.containsKey(key)) {
77        Term t1 = terms.get(key);
78        if (!t1.equals(term)) {
79          LOG
80            .warn("Terms {} and {} are both known as \"{}\". Keeping only {}", terms.get(key), term, key, terms.get(key));
81        }
82      } else {
83        terms.put(key, term);
84      }
85    }
86  
87    /**
88     * @return a purely alphanumerical, lower cased term with all other characters replaced
89     */
90    public static String normaliseTerm(String term) {
91      return NON_ALPHA_NUM_PATTERN.matcher(term).replaceAll("").toLowerCase();
92    }

94    /**
95     * This is the main method to get a term from the factory.
96     * It will lookup matching terms applying some normalization and known synonyms first.
97     * If nothing matches the factory creates a new UnknownTerm instance and keeps it for further requests so that
98     * all terms with the same qualified name return a single UnknownTerm instance.
99     *
100    * For clearly bad term names an IllegalArgumentException is thrown.
101    * For example in the case of a simple name containing whitespace like "hello tom".
102    * Ideally the term names to be looked up should be full URIs, but simple names made up of alphanumerics and dashes
103    * will also work fine. Unknown simple names will be put into the namespace http://unknown.org when a new UnknownTerm
104    * instance is created.
105    */
106   public Term findTerm(String termName) throws IllegalArgumentException {
107     if (termName == null || termName.trim().isEmpty()) {
108       return null;
109     }
110     // normalise terms
111     Term term;
112     String normTermName = normaliseTerm(termName);
113     // try known terms first
114     if (terms.containsKey(normTermName)) {
115       return terms.get(normTermName);
116     } else {
117       // create new term instance
118       try {
119         term = UnknownTerm.build(termName);
120       } catch (IllegalArgumentException e) {
121         // simple names as found in ATB file headers are rejected
122         // convert into a standard unknown term namespace and try again
123         term = UnknownTerm.build(UNKNOWN_NAMESPACE + termName);
124       }
125       addTerm(normTermName, term);
126       addTerm(termName, term);
127     }
128     return term;
129   }
