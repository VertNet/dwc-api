254   private DwcTerm(String groupName, String... alternatives) {
255     normQName = TermFactory.normaliseTerm(qualifiedName());
256     for (int i = 0; i < alternatives.length; i++) {
257       alternatives[i] = TermFactory.normaliseTerm(alternatives[i]);
258     }
259     normAlts = alternatives;
260     this.groupName = groupName;
261   }

305   /**
306    * @return true if the dwc term is defining a class instead of a property, e.g. Taxon
307    */
308   public boolean isClass() {
309     return Character.isUpperCase(simpleName().charAt(0));
310   }

312   /**
313    * List all terms that belong to a given group.
314    *
315    * @param group the group to list terms for
316    *
317    * @return the list of dwc terms in the given group
318    */
319   public static List<DwcTerm> listByGroup(String group) {
320     List<DwcTerm> terms = new ArrayList<DwcTerm>();
321     for (DwcTerm t : DwcTerm.values()) {
322       if (t.getGroup().equalsIgnoreCase(group)) {
323         terms.add(t);
324       }
325     }
326     return terms;
327   }
